class Box {
    double width, height, depth;
    Box() {
        width = 10;
        height = 10;
        depth = 10;
    }
    double volume() {
        return width * height * depth;
    }
}

public class BoxVolume {
    public static void main(String[] args) {
        Box box = new Box();
        double vol = box.volume();
        System.out.println("Volume of the box: " + vol);
    }
}
